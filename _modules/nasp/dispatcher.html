import logging
import job


def _parse_args():
    import argparse

    parser = argparse.ArgumentParser(description="Meant to be called from the pipeline automatically.")
    parser.add_argument("--config", required=True, help="Path to the configuration xml file.")
    return parser.parse_args()


def _submit_job(job_submitter, command, job_parms, waitfor_id=None, hold=False, notify=False):
    import subprocess
    import re
    import os

    # TODO(jtravis): remove unused output variable
    output = jobid = None
    logging.info("command = %s" % command)
    if job_submitter == "PBS":
        waitfor = ""
        if waitfor_id:
            dependency_string = waitfor_id[1] if len(waitfor_id) > 1 else 'afterok'
            waitfor = "-W depend=%s:%s" % (dependency_string, waitfor_id[0])
        queue = ""
        if job_parms["queue"]:
            queue = "-q %s" % job_parms["queue"]
        args = job_parms["args"]
        if hold:
            args += " -h"
        if notify:
            args += " -m e"
        submit_command = "qsub -d \'%s\' -w \'%s\' -l ncpus=%s,mem=%sgb,walltime=%s:00:00 -m a -N \'%s\' %s %s %s" % (
            job_parms["work_dir"], job_parms["work_dir"], job_parms['num_cpus'], job_parms['mem_requested'],
            job_parms['walltime'], job_parms['name'], waitfor, queue, args)
        logging.debug("submit_command = %s", submit_command)
        output = subprocess.getoutput("echo \"%s\" | %s - " % (command, submit_command))
        logging.debug("output = %s" % output)
        job_match = re.search('^(\d+)\..*$', output)
        if job_match:
            jobid = job_match.group(1)
        else:
            logging.warning("Job not submitted!!")
            print("WARNING: Job not submitted: %s" % output)
    elif job_submitter == "SLURM":
        waitfor = ""
        if waitfor_id:
            dependency_string = waitfor_id[1] if len(waitfor_id) > 1 else 'afterok'
            waitfor = "-d %s:%s" % (dependency_string, waitfor_id[0])
        queue = ""
        if job_parms["queue"]:
            queue = "-p %s" % job_parms["queue"]
        args = job_parms["args"]
        if hold:
            args += " -H"
        if notify:
            args += " --mail-type=END"
        submit_command = "sbatch -D \'%s\' -c%s --mem=%s000 --mail-type=FAIL -J \'%s\' %s %s %s" % (
            job_parms["work_dir"], job_parms['num_cpus'], job_parms['mem_requested'], job_parms['name'], waitfor, queue,
            args)
        logging.debug("submit_command = %s" % submit_command)
        output = subprocess.getoutput("%s --wrap=\"%s\"" % (submit_command, command))
        logging.debug("output = %s" % output)
        job_match = re.search('^Submitted batch job (\d+)$', output)
        if job_match:
            jobid = job_match.group(1)
        else:
            logging.warning("Job not submitted!!")
            print("WARNING: Job not submitted: %s" % output)
    elif job_submitter == "SGE":
        waitfor = ""
        if waitfor_id:
            waitfor = "-hold_jid %s" % (re.sub(":", ",", waitfor_id[0]))
        queue = ""
        if job_parms["queue"]:
            queue = "-q %s" % job_parms["queue"]
        args = job_parms["args"]
        if hold:
            args += " -h"
        if notify:
            args += " -m e"
        mem_needed = float(job_parms['mem_requested']) * 1024 * 1024
        # Apparently the number of processors a job uses is controlled by the queue it is running on in SGE, so there is no way to request a specific number of CPUs??
        submit_command = "qsub -cwd \'%s\' -wd \'%s\' -l h_data=%sgb,h_rt=%s:00:00 -m a -N \'%s\' %s %s %s" % (
            job_parms["work_dir"], job_parms["work_dir"], mem_needed, job_parms['walltime'], job_parms['name'], waitfor,
            queue, args)
        logging.debug("submit_command = %s", submit_command)
        output = subprocess.getoutput("echo \"%s\" | %s - " % (command, submit_command))
        logging.debug("output = %s" % output)
        job_match = re.search('^(\d+)\..*$', output)
        if job_match:
            jobid = job_match.group(1)
        else:
            logging.warning("Job not submitted!!")
            print("WARNING: Job not submitted: %s" % output)
    else:
        command = re.sub('\n', '; ', command)
        work_dir = job_parms['work_dir']
        dependency_check = ""
        if waitfor_id:
            if re.search(":", waitfor_id[0]):
                pid_filename = os.path.join(work_dir, "%s_dependent_pids" % job_parms['name'])
                pid_file = open(pid_filename, 'w')
                pids = waitfor_id[0].split(":")
                pid_file.write("\n".join(pids))
                pid_file.close()
                dependency_check = "while [ -s %s ]; do sleep 600; for pid in `cat %s`; do kill -0 \"$pid\" 2>/dev/null || sed -i \"/^$pid$/d\" %s; done; done; rm %s; " % (
                    pid_filename, pid_filename, pid_filename, pid_filename)
            else:
                pid = waitfor_id[0]
                dependency_check = "while kill -0 %s; do sleep 300; done; " % pid
        # cpu_check = "grep 'cpu ' /proc/stat | awk '{usage=($2+$4)/($2+$4+$5)} END {print 1-usage}'"
        total_mem = subprocess.getoutput("free -g | grep Mem: | awk '{ print $2 }'")
        mem_requested = job_parms['mem_requested']
        if mem_requested > total_mem:
            mem_requested = total_mem
        print("total_mem = %s" % total_mem)
        mem_needed = float(mem_requested) * 750
        mem_check = "while [ `free -m | grep cache: | awk '{ print $4 }'` -lt %d ]; do sleep 300; done; " % mem_needed
        submit_command = "%s%s%s" % (dependency_check, mem_check, command)
        logging.debug("submit_command = %s" % submit_command)
        output_log = os.open(os.path.join(work_dir, "%s.out" % job_parms['name']), os.O_WRONLY | os.O_CREAT)
        proc = subprocess.Popen(submit_command, stderr=subprocess.STDOUT, stdout=output_log, shell=True, cwd=work_dir)
        jobid = str(proc.pid)
    logging.info("jobid = %s" % jobid)
    return jobid


def _release_hold(job_submitter, job_id):
    import subprocess

    if job_submitter == "PBS" or job_submitter == "SGE":
        command = "qrls %s" % job_id
    elif job_submitter == "SLURM":
        command = "scontrol release %s" % job_id
    else:
        return
    logging.info("command = %s", command)
    output = subprocess.getoutput(command)
    logging.debug("output = %s", output)


def _align_reads(read, configuration, index_job_id, reference):
    """
    Args:
        read_tuple (ReadFile):
        configuration (dict):
        index_job_id:
        reference (str):

    Returns:
        [(bam_nickname, job_id, final_file, aligner, snpcaller)]:
    """
    import re
    aligner_output = []
    for aligner in configuration["aligners"]:
        if re.search('bwa', aligner.name, re.IGNORECASE):
            bwa_job = job.Bwa(read, aligner, configuration["samtools"], reference, configuration["output_folder"])
            job_id = _submit_job(configuration["job_submitter"], bwa_job.command, bwa_job.parameters, (index_job_id,))

            if job_id:
                aligner_output.append((bwa_job.bam_nickname, job_id, bwa_job.final_file, aligner.name))
        elif re.search('b(ow)?t(ie)?2', aligner.name, re.IGNORECASE):
            bowtie2_job = job.Bowtie2(read, aligner, configuration["samtools"], reference, configuration["output_folder"])
            job_id = _submit_job(configuration["job_submitter"], bowtie2_job.command, bowtie2_job.parameters, (index_job_id,))

            if job_id:
                aligner_output.append((bowtie2_job.bam_nickname, job_id, bowtie2_job.final_file, aligner.name))
        elif re.search('novo', aligner.name, re.IGNORECASE):
            novo_job = job.Novoalign(read, aligner, configuration["samtools"], reference, configuration["output_folder"])
            job_id = _submit_job(configuration["job_submitter"], novo_job.command, novo_job.parameters, (index_job_id,))

            if job_id:
                aligner_output.append((novo_job.bam_nickname, job_id, novo_job.final_file, aligner.name))
        elif re.search('snap', aligner.name, re.IGNORECASE):
            snap_job = job.Snap(read, aligner, configuration["samtools"], reference, configuration["output_folder"])
            job_id = _submit_job(configuration["job_submitter"], snap_job.command, snap_job.parameters, (index_job_id,))

            if job_id:
                aligner_output.append((snap_job.bam_nickname, job_id, snap_job.final_file, aligner.name))
        else:
            print("Unknown aligner \'%s\' found, don't know what to do. Skipping..." % aligner.name)
    return aligner_output


def _call_snps(aligner_output, configuration, reference):
    """
    Args:
        aligner_output (list):
        configuration (dict):
        reference (str):

    Returns:
        [(str, str, str, str, str)]: (vcf_nickname, job_id, final_file, aligner, snpcaller)
    """
    import re

    snpcaller_output = []
    for (nickname, aligner_job_id, bam_file, aligner_name) in aligner_output:
        if aligner_job_id:
            for snpcaller in configuration["snpcallers"]:
                name = snpcaller[0]
                if re.search('gatk', name, re.IGNORECASE):
                    gatk_job = job.Gatk(nickname, bam_file, snpcaller, reference, configuration["output_folder"])
                    job_id = _submit_job(configuration["job_submitter"], gatk_job.command, gatk_job.parameters, (aligner_job_id,))

                    if job_id:
                        snpcaller_output.append((gatk_job.vcf_nickname, job_id, gatk_job.final_file, aligner_name, name))
                elif re.search('solsnp', name, re.IGNORECASE):
                    solsnp_job = job.Solsnp(nickname, bam_file, snpcaller, reference, configuration["output_folder"])
                    job_id = _submit_job(configuration["job_submitter"], solsnp_job.command, solsnp_job.parameters, (aligner_job_id,))

                    if job_id:
                        snpcaller_output.append((solsnp_job.vcf_nickname, job_id, solsnp_job.final_file, aligner_name, name))
                elif re.search('varscan', name, re.IGNORECASE):
                    varscan_job = job.Varscan(configuration["samtools"], nickname, bam_file, snpcaller, reference, configuration["output_folder"])
                    job_id = _submit_job(configuration["job_submitter"], varscan_job.command, varscan_job.parameters, (aligner_job_id,))

                    if job_id:
                        snpcaller_output.append((varscan_job.vcf_nickname, job_id, varscan_job.final_file, aligner_name, name))
                elif re.search('samtools', name, re.IGNORECASE):
                    samtools_job = job.Samtools(configuration["samtools"], nickname, bam_file, snpcaller, reference, configuration["output_folder"])
                    job_id = _submit_job(configuration["job_submitter"], samtools_job.command, samtools_job.parameters, (aligner_job_id,))

                    if job_id:
                        snpcaller_output.append((samtools_job.vcf_nickname, job_id, samtools_job.final_file, aligner_name, name))
                else:
                    print("Unknown SNP caller \'%s\' found, don't know what to do. Skipping..." % name)
    return snpcaller_output

[docs]def begin(configuration):
    """
    Args:
        configuration (dict):
    """
    index_job = job.IndexReference(configuration['aligners'], configuration['snpcallers'], configuration['samtools'], configuration['picard'], configuration['index'], configuration['reference'].path, configuration['output_folder'])
    index_job_id = _submit_job(configuration["job_submitter"], index_job.command, index_job.parameters, hold=True)
    reference = index_job.final_file

    if not index_job_id:
        print("Failed to submit the index job, there is no point in continuing. Please try again.")
        raise SystemExit()
    dups_file = None
    job_ids = []
    vcf_files = []
    franken_fastas = []
    if configuration["find_dups"]:
        find_dups_job = job.FindDups(configuration["dup_finder"], reference)
        dups_file = find_dups_job.final_file
        job_id = _submit_job(configuration["job_submitter"], find_dups_job.command, find_dups_job.parameters, (index_job_id,))
        if job_id:
            job_ids.append(job_id)
    for assembly in configuration["assemblies"]:
        external_genome_job = job.ExternalGenome(assembly, configuration["dup_finder"], configuration["assembly_importer"], reference, configuration["output_folder"])
        job_id = _submit_job(configuration["job_submitter"], external_genome_job.command, external_genome_job.parameters, (index_job_id,))
        if job_id:
            job_ids.append(job_id)
            franken_fastas.append((assembly.sample, "nucmer", external_genome_job.final_file))
    if configuration["alignments"]:
        pre_aligned = []
        index_bam_job = job.IndexBam(configuration['bam_index'], configuration['samtools'], configuration['alignments'], configuration['output_folder'])
        bamindex_job_id = _submit_job(configuration["job_submitter"], index_bam_job.command, index_bam_job.parameters, (index_job_id,))
        for bam_file in index_bam_job.bam_files:
            pre_aligned.append((bam_file.name, bamindex_job_id, bam_file.path, "pre-aligned"))
        snpcaller_output = _call_snps(pre_aligned, configuration, reference)
        for (vcf_nickname, job_id, final_file, aligner, snpcaller) in snpcaller_output:
            if job_id:
                job_ids.append(job_id)
                vcf_files.append((vcf_nickname, aligner, snpcaller, final_file))
    for read_tuple in configuration["reads"]:
        aligner_output = _align_reads(read_tuple, configuration, index_job_id, reference)
        snpcaller_output = _call_snps(aligner_output, configuration, reference)
        for (vcf_nickname, job_id, final_file, aligner, snpcaller) in snpcaller_output:
            if job_id:
                job_ids.append(job_id)
                vcf_files.append((vcf_nickname, aligner, snpcaller, final_file))
    for (name, vcf) in configuration["vcfs"]:
        vcf_files.append((name, "pre-aligned", "pre-called", vcf))

    matrix_job = job.Matrix(dups_file, vcf_files, franken_fastas, configuration['matrix_generator'], configuration['coverage_filter'], configuration['proportion_filter'], configuration['filter_matrix_format'], job_ids, reference, configuration['output_folder'])
    job_id = _submit_job(configuration["job_submitter"], matrix_job.command, matrix_job.parameters, (matrix_job.jobs_to_wait_for, 'afterany'), notify=True)
    _release_hold(configuration["job_submitter"], index_job_id)
