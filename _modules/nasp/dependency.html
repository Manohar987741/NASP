[docs]def find_executable(executable):
    """
    Returns the path to the executable.

    Args:
        executable (str): executable to find

    Returns:
        str: Path to the executable or an empty string

    Raises:
        UnlistedDependencyError: The executable is not in the list of dependencies
    """
    path = None

    # Warning for developers
    if executable not in paths:
        raise UnlistedDependencyError(executable)

    try:
        # shutil requires Python >= 3.3
        import shutil

        mode=os.F_OK | os.X_OK
        # .jar files do not need the executable bit set
        if executable.endswith(".jar"):
            mode=os.F_OK

        # Search based on the PATH or platform dependent default environment variables
        path = shutil.which(executable)

        # Search for .jars in the CLASSPATH and /usr/share/java/
        if path is None and executable.endswith(".jar"):
            path = shutil.which(executable, mode=mode, path=":".join(['/usr/share/java/', os.getenv("CLASSPATH")]) if os.getenv("CLASSPATH") else '/usr/share/java/')
    except ImportError:
        try:
            import subprocess
            path = subprocess.check_output(["which", executable])
        except subprocess.CalledProcessError:
            # File not found, try something else
            pass

    # The executable may have been found during a recursive search in a previous call to this method
    # The cache is checked after the PATH is searched because PATH is considered the canonical source
    if path is None:
        path = paths[executable]

    # FIXME: Prompt the user where / if they would like to recursively search for the executable
    # to avoid repeating the search for them as well
    # Recursively search the user's home directory
    if path is None:
        for root, dirnames, filenames in os.walk(os.path.expanduser('~')):
            for command, path in paths.items():
                if path is None and command in filenames:
                    return os.path.join(root, command)

    paths[executable] = path

    return path or ""



[docs]class UnlistedDependencyError(Exception):
    """Warning for developers if they forget to add a dependency to the dictionary of paths

    Attributes:
        executable (str): Name of the missing dependency
    """

    def __init__(self, executable):
        self.executable = executable

    def __str__(self):
        return "Did you forget to add %s as a dependency?" % repr(self.executable)
