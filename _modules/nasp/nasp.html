def _expand_path(path):
    """
    Args:
        path (str): filename path to expand

    Returns:
        str: absolute filepath with any variables expanded
    """
    return os.path.abspath(os.path.expanduser(path))


def _find_files(path, extension):
    """
    Return all files in path with the given extension

    Args:
        path (str): Directory to search for files
        extension (str): Extension of files to match

    Returns:
        list of NaspFile:
    """
    # TODO compare glob vs os.walk
    import glob

    file_list = []
    for filepath in glob.glob(os.path.join(path, "*." + extension)):
        # sample_name is the filename without the extension
        sample_name = os.path.splitext(os.path.basename(filepath))[0]
        nasp_file = NaspFile(sample=sample_name, path=filepath)
        file_list.append(nasp_file)
        logging.info(nasp_file)
    return file_list


def _find_reads(path):
    """
    Returns the fasta files found under path

    Args:
        path (str): Directory to search for fasta files

    Returns:
        list of ReadFile: Fasta files found under path
    """
    import re

    read_list = []
    for file in os.listdir(path):
        is_read = re.search('(.*)(\.fastq(?:\.gz)?)$', file, re.IGNORECASE)
        if is_read:
            sample_name = is_read.group(1)
            is_paired = re.search('^(.*)(_[R]?)([12])(.*)$', sample_name, re.IGNORECASE)
            if is_paired:
                if is_paired.group(3) == '1':  # If paired, only process read 1, so we don't double count the pair
                    sample_name = is_paired.group(1)
                    read1 = file
                    read2 = "%s%s2%s%s" % (is_paired.group(1), is_paired.group(2), is_paired.group(4), is_read.group(2))
                    if os.path.exists(os.path.join(path, read2)):
                        #read = (sample_name, os.path.join(path, read1), os.path.join(path, read2))
                        read = ReadFile(
                            name=sample_name,
                            path=os.path.join(path, read1),
                            path2=os.path.join(path, read2)
                        )
                        read_list.append(read)
                        logging.info(read)
                    else:
                        logging.warning("Cannot find %s, the matching read to %s. Skipping..." % (read2, read1))
            else:
                #read = (sample_name, os.path.join(path, file))
                read = ReadFile(
                    name=sample_name,
                    path=os.path.join(path, read1),
                    path2=""
                )
                read_list.append(read)
                logging.info(read)
    return read_list


def _get_bams(cwd):
    """
    Returns the .bam and .sam files found in cwd or the path specified by the user.

    Args:
        cwd (str): Default directory to search for .bam files

    Returns:
        list of NaspFile: bam and sam files found
    """
    import re

    bam_list = []
    response = input("\nDo you have pre-aligned SAM/BAM files you wish to include [N]? ")
    if re.match('^[Yy]', response):
        path = input("Where are these files located [%s]? " % cwd)
        path = _expand_path(path) if path else cwd
        logging.info("Looking for sams/bams in %s...", path)
        bam_list = _find_files(path, "bam")
        bam_list.extend(_find_files(path, "sam"))
    return bam_list

def _get_external_fastas(cwd, exclude):
    """
    Returns a list of NaspFile objects representing the external fasta files found

    Args:
        cwd (str): Default directory to search
        exclude (str):

    Returns:
        list: List of NaspFile
    """
    import re

    fasta_list = []
    response = input("\nDo you have fasta files for external genomes you wish to include [Y]? ")
    if not re.match('^[Nn]', response):
        path = input("Where are these files located [%s]? " % cwd)
        path = _expand_path(path) if path else cwd
        logging.info("Looking for external fastas in %s...", path)
        fasta_list = _find_files(path, "fasta")
        # remove the reference from the fasta_list if it is in there
        for fasta_tuple in fasta_list:
            fasta = fasta_tuple[1]
            if fasta == exclude:
                fasta_list.remove(fasta_tuple)
    return fasta_list

def _get_application_path(filename):
    """
    Args:
        filename (str): application to find

    Returns:
        str: Path to the application
    """
    app_path = dependency.find_executable(filename)

    if not app_path:
        app_path = input("\nUnable to find '%s', please enter the full path: " % filename)

    # Verify the file exists and is executable
    mode=os.F_OK | os.X_OK
    if app_path.endswith(".jar"):
        # .jar files do not require the executable bit
        mode=os.F_OK

    while not os.access(app_path, mode):
        app_path = input(
            "\n'%s' either does not exist or you don't have permission to run it, please enter the full path to '%s': " % (
            app_path, filename))
        # Verify the file exists and is executable
        mode=os.F_OK | os.X_OK
        if app_path.endswith(".jar"):
            # .jar files do not require the executable bit
            mode=os.F_OK
    return app_path


def _get_advanced_settings(defaults):
    """
    Args:
        defaults (Application): Default settings if the user does not define them

    Returns:
        Application: The application settings
    """
    import re

    response = input("Would you like to set advanced %s settings [N]? " % defaults.name)
    if re.match('^[Yy]', response):
        alt_version = input("  Would you like to use an alternate %s version [N]? " % defaults.name)
        if re.match('^[Yy]', alt_version):
            path = input("    What is the full path to the %s runtime you wish to use [%s]? " % (defaults.name, defaults.path))
            path = path if path else defaults.path
            while not os.access(path, os.X_OK):
                path = input(
                    "    '%s' either does not exist or you don't have permission to run it, please enter the full path to '%s': " % (
                    path, defaults.name))
                #
                path = defaults.path
            defaults._replace(path=path)
        args = input("  What additional arguments would you like to pass to %s [%s]? " % (defaults.name, defaults.additional_args))
        if args:
            defaults._replace(additional_args=args)
        queue = input("  What queue/partition should %s run on [%s]? " % (defaults.name, defaults.job_parameters['queue']))
        if queue:
            defaults.job_parameters['queue'] = queue
        mem = input("  How much memory (GB) will %s require [%s]? " % (defaults.name, defaults.job_parameters['mem_requested']))
        if re.match("^[1-9][0-9]*$", mem):
            defaults.job_parameters['mem_requested'] = mem
        cpus = input("  How many CPUs do you want %s to use [%s]? " % (defaults.name, defaults.job_parameters['num_cpus']))
        if re.match("^[1-9][0-9]*$", cpus):
            defaults.job_parameters['num_cpus'] = cpus
        hours = input("  How many hours will %s take to run [%s]? " % (defaults.name, defaults.job_parameters['walltime']))
        if re.match("^[1-9][0-9]*$", hours):
            defaults.job_parameters['walltime'] = hours

    return defaults

def _get_job_submitter():
    import re

    job_submitter = "invalid"
    queue = ""
    args = ""
    response = input(
        "\nWhat system do you use for job management (PBS/TORQUE, SLURM, SGE*, and 'none' are currently supported) [PBS]? ")
    while job_submitter == "invalid":
        if re.match('^(PBS|Torque|qsub)$', response, re.IGNORECASE) or response == "":
            job_submitter = "PBS"
        elif re.match('^(SLURM|sbatch)$', response, re.IGNORECASE):
            job_submitter = "SLURM"
        elif re.match('^SGE', response, re.IGNORECASE):
            job_submitter = "SGE"
        elif re.match('^none$', response, re.IGNORECASE):
            job_submitter = "NONE"
        else:
            response = input("  %s is not a valid job management system, please enter another [PBS]? " % response)
    if job_submitter != "NONE":
        queue = input(
            "  Would you like to specify a queue/partition to use for all jobs (leave blank to use default queue) []? ")
        args = input("  What additional arguments do you need to pass to the job management system []? ")
    return job_submitter, queue, args


def _get_user_input(reference, output_folder):

        output_folder = input("\nWhere would you like output files to be written [nasp_results]? ")

            output_folder = "nasp_results"

        if not re.match('^[Yy]', response):
            print("Operation cancelled!")
            quit()

        os.makedirs(output_folder)

                        format='%(asctime)s %(levelname)-8s %(message)s',

                        filemode='w')

    if not reference:
        reference = input("\nWhere is the reference fasta file you would like to use? ")

    import nasp.configuration_parser as configuration_parser
    import re

    commandline_args = _parse_args()
    if commandline_args.config:
        configuration = configuration_parser.parse_config(commandline_args.config)
        output_folder = configuration["output_folder"]
        if os.path.exists(output_folder):
            response = input(
                "\nOutput folder %s already exists!\nFiles in it may be overwritten!\nShould we continue anyway [N]? " % output_folder)
            if not re.match('^[Yy]', response):
                print("Operation cancelled!")
                quit()
        else:
            os.makedirs(output_folder)
        logfile = os.path.join(output_folder, "runlog.txt")
        logging.basicConfig(level=logging.DEBUG,
                            format='%(asctime)s %(levelname)-8s %(message)s',
                            datefmt='%m/%d/%Y %H:%M:%S',
                            filename=logfile,
                            filemode='w')
    else:
        configuration = _get_user_input(commandline_args.reference_fasta, commandline_args.output_folder)
    configuration_parser.write_config(configuration)
    dispatcher.begin(configuration)



if __name__ == "__main__":
    main()
